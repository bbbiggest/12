## HDU1285-确定比赛名次 题解 @author bbig

用一个数组记录入度，然后每次将当前入度为0的且序号最小的顶点输出即可。

要注意这题应该是有重边的 用邻接表就没问题 用邻接矩阵就可能会错 假如用邻接矩阵的话 可以先判断如果两条边已经连通 就不用再增加入度 或者是后续删边的时候要记得入度也减少相应的数量。

还要注意这题输出最后面没有空格，否则会格式错误。

**邻接表：**


``` c++
#include <iostream>
#include <cstring>
#include <vector>
using namespace std;
const int MAX_V = 510;

vector<int> G[MAX_V];
int V, E;
int in[MAX_V]; // 记录顶点的入度

void toposort()
{
    for (int cnt = 0; cnt < V; ++cnt)
    {
        int i;
        for (i = 1; in[i] != 0; ++i) // 找到当前入度为0的最小序号
            ;
        if (cnt == 0) // 要注意这道题输出最后面没有空格
            cout << i;
        else
            cout << " " << i;
        in[i] = -1;
        for (int j = 0; j < G[i].size(); ++j) // 将相邻的顶点入度减1
            in[G[i][j]]--;
    }
    cout << "\n";
}

int main()
{
    while (cin >> V >> E)
    {
        memset(G, false, sizeof(G));
        memset(in, 0, sizeof(in));
        for (int i = 0; i < E; ++i) // 邻接表储存
        {
            int a, b;
            cin >> a >> b;
            G[a].push_back(b);
            in[b]++;
        }
        toposort();
    }
    return 0;
}

```

**邻接矩阵：**（全部替换为注释的代码也是一样的）

``` c++
#include <iostream>
#include <cstring>
using namespace std;
const int MAX_V = 510;

bool G[MAX_V][MAX_V]; // int G[MAX_V][MAX_V];
int V, E;
int in[MAX_V];

void toposort()
{
    for (int k = 1; k <= V; ++k)
    {
        int i;
        for (i = 1; in[i] != 0; ++i)
            ;
        if (k == 1)
            cout << i;
        else
            cout << " " << i;
        in[i] = -1;
        for (int j = 1; j <= V; ++j)
        {
            if (G[i][j])
                in[j]--; // in[j] -= G[i][j];
        }
    }
    cout << "\n";
}

int main()
{
    while (cin >> V >> E)
    {
        memset(G, 0, sizeof(G));
        memset(in, 0, sizeof(in));

        for (int i = 0; i < E; ++i)
        {
            int a, b;
            cin >> a >> b;
            
            if (G[a][b] == false)
            {
                G[a][b] = true;
                in[b]++;
            }
            // 上面的if的5行可替换为
            // G[a][b]++;
            // in[b]++;
            
        }
        toposort();
    }
    return 0;
}
```
